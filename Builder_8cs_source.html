55 public void Build()
56 {
57 _state = State.Started;
58 Log.Info("Build start");
59
60 _state = State.CollectingSourceItems;
61 while (_sources.TryPop(out var source))
62 {
63 try
64 {
65 source.Custom(_query);
66 var items = source.CollectItems();
67 _sourceItems.AddRange(items);
68 }
69 catch (Exception e)
70 {
71 throw new Exception($"Source {source.Name} failed.", e);
72 }
73 }
74
75 _state = State.ConvertingSourceItems;
76 while (_sourceItemConverters.TryPop(out var converter))
77 {
78 try
79 {
80 var items = converter.Convert(_sourceItems, _query);
81 _sourceItems.Clear();
82 _sourceItems.AddRange(items);
83 }
84 catch (Exception e)
85 {
86 throw new Exception($"Source item converter {converter.Name} failed.", e);
87 }
88 }
89
90 _state = State.FormattingSourceItems;
91 while (_formatters.TryPop(out var formatter))
92 {
93 try
94 {
95 var content = formatter.Format(_sourceItems, _query);
96 _contents.Add(content);
97 }
98 catch (Exception e)
99 {
100 throw new Exception($"Formatter {formatter.Name} failed.", e);
101 }
102 }
103
104 _state = State.ConvertingContents;
105 while (_contentConverters.TryPop(out var converter))
106 {
107 try
108 {
109 var contents = converter.Convert(_contents, _query);
110 _contents.Clear();
111 _contents.AddRange(contents);
112 }
113 catch (Exception e)
114 {
115 throw new Exception($"Content converter {converter.Name} failed.", e);
116 }
117 }
118
119 _state = State.WritingContents;
120 while (_writers.TryPop(out var writer))
121 {
122 try
123 {
124 writer.Write(_contents, _query);
125 }
126 catch (Exception e)
127 {
128 throw new Exception($"Writer {writer.Name} failed.", e);
129 }
130 }
131
132 _state = State.Completed;
133 Log.Info("Build completed.");
134 }

151 public void Use<T>(T instance) where T : class
152 {
153 switch (instance)
154 {
155 case ISource source:
156 if (_state > State.CollectingSourceItems)
157 {
158 throw new InvalidOperationException("Cannot add source after collecting source items.");
159 }
160 _sources.Push(source, source.Priority, source.Name);
161 break;
162 case IFormatter formatter:
163 if (_state > State.FormattingSourceItems)
164 {
165 throw new InvalidOperationException("Cannot add formatter after formatting source items.");
166 }
167 _formatters.Push(formatter, formatter.Priority, formatter.Name);
168 break;
169 case IWriter writer:
170 if (_state > State.WritingContents)
171 {
172 throw new InvalidOperationException("Cannot add writer after writing contents.");
173 }
174 _writers.Push(writer, writer.Priority, writer.Name);
175 break;
176 case ISourceItemConverter sourceItemConverter:
177 if (_state > State.ConvertingSourceItems)
178 {
179 throw new InvalidOperationException("Cannot add source item converter after converting source items.");
180 }
181 _sourceItemConverters.Push(sourceItemConverter, sourceItemConverter.Priority, sourceItemConverter.Name);
182 break;
183 case IContentConverter contentConverter:
184 if (_state > State.ConvertingContents)
185 {
186 throw new InvalidOperationException("Cannot add content converter after converting content.");
187 }
188 _contentConverters.Push(contentConverter, contentConverter.Priority, contentConverter.Name);
189 break;
190 default:
191 throw new ArgumentException("Invalid type", nameof(T));
192 }
193 }

302 if (_names.Contains(name))
303 {
304 throw new ArgumentException($"Duplicate {_containerType.ToString()}: {name}");
