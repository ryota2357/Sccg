53 public void Build()
54 {
55 _state = State.Started;
56 Log.Info("Build start");
57
58 _state = State.CollectingSourceItems;
59 while (_sources.TryPop(out var source))
60 {
61 try
62 {
63 source.Custom(_query);
64 var items = source.CollectItems();
65 _sourceItems.AddRange(items);
66 }
67 catch (Exception e)
68 {
69 throw new Exception($"Source {source.Name} failed.", e);
70 }
71 }
72
73 _state = State.FormattingSourceItems;
74 while (_formatters.TryPop(out var formatter))
75 {
76 try
77 {
78 var content = formatter.Format(_sourceItems, _query);
79 _contents.Add(content);
80 }
81 catch (Exception e)
82 {
83 throw new Exception($"Formatter {formatter.Name} failed.", e);
84 }
85 }
86
87 _state = State.WritingContents;
88 while (_writers.TryPop(out var writer))
89 {
90 try
91 {
92 writer.Write(_contents, _query);
93 }
94 catch (Exception e)
95 {
96 throw new Exception($"Writer {writer.Name} failed.", e);
97 }
98 }
99
100 _state = State.Completed;
101 Log.Info("Build completed.");
102 }

117 public void Use<T>(T instance) where T : class
118 {
119 switch (instance)
120 {
121 case ISource source:
122 if (_state > State.CollectingSourceItems)
123 {
124 throw new InvalidOperationException("Cannot add source after collecting source items.");
125 }
126 _sources.Push(source, source.Priority, source.Name);
127 break;
128 case IFormatter formatter:
129 if (_state > State.FormattingSourceItems)
130 {
131 throw new InvalidOperationException("Cannot add formatter after formatting source items.");
132 }
133 _formatters.Push(formatter, formatter.Priority, formatter.Name);
134 break;
135 case IWriter writer:
136 if (_state > State.WritingContents)
137 {
138 throw new InvalidOperationException("Cannot add writer after writing contents.");
139 }
140 _writers.Push(writer, writer.Priority, writer.Name);
141 break;
142 default:
143 throw new ArgumentException("Invalid type", nameof(T));
144 }
145 }
