24 protected override SingleTextContent Format(IEnumerable<IVimSourceItem> items, BuilderQuery query)

26 var metadata = query.GetMetadata();

28 var header = metadata.Header(metadata);
29 if (header is null)
30 {
31 var data = new[]
32 {
33 ("Name", metadata.ThemeName),
34 ("Version", metadata.Version),
35 ("Author", metadata.Author),
36 ("Maintainer", metadata.Maintainer),
37 ("License", metadata.License),
38 ("Description", metadata.Description),

41 ("Last change", metadata.LastUpdated?.ToString("yyyy-MM-dd dddd", System.Globalization.CultureInfo.CreateSpecificCulture("en-US")))

43 .OfType<(string, string)>()
44 .ToArray();
45 if (data.Length != 0)
46 {
47 var maxLen = data.Max(x => x.Item1.Length);
48 header = data.Select(x => $"{(x.Item1 + ":").PadRight(maxLen + 1, ' ')} {x.Item2}").ToArray();
49 }
50 else

53 }
54 }

56

58 foreach (var item in items)
59 {
60 var formattable = item.Extract();

62 if (formattable.Link is not null)

64 body.Add($"hi link {formattable.Name} {formattable.Link}");

68 var sb = new StringBuilder();
69
70 void Set<T>(string name, T? value)
71 {
72 if (value is null) return;
73 switch (value)
74 {
75 case Color c:
76 if (c.IsDefault) return;
77 var code = c.IsNone ? "NONE" : c.HexCode;
78 sb.Append($" {name}={code}");
79 break;
80 case string s:
81 sb.Append($" {name}={s}");
82 break;
83 default:
84 throw new NotSupportedException($"NeovimFormatter does not support type {typeof(T).Name}");
85 }
86 }
87
88 sb.Append($"hi {formattable.Name}{(formattable.Default ? " default" : "")}");
89 Set("ctermfg", formattable.Style?.Foreground.TerminalColorCode);
90 Set("ctermbg", formattable.Style?.Background.TerminalColorCode);
91 Set("ctermul", formattable.Style?.Special.TerminalColorCode);
92 Set("cterm", CreateAttrList(formattable.Style?.Modifiers));
93 Set("guifg", formattable.Style?.Foreground);
94 Set("guibg", formattable.Style?.Background);
95 Set("guisp", formattable.Style?.Special);
96 Set("gui", CreateAttrList(formattable.Style?.Modifiers));
97 body.Add(sb.ToString());
98 }
99
100 var footer = metadata.Footer(metadata);
101 if (footer is null)
102 {
103 footer = new[] { $"Built with Sccg {Metadata.__SccgVersion}" };
104 }
105
106 return new SingleTextContent($"colors/{metadata.ThemeName}.vim",
107 string.Join('\n', header.Select(x => $"\" {x}")),
108 $"""
109 hi clear
110 if exists('syntax_on')
111 syntax reset
112 endif
113 let g:colors_name = '{metadata.ThemeName ?? "sccg_default"}'
114 """,
115 string.Join('\n', body),
116 string.Join('\n', footer.Select(x => $"\" {x}"))
117 );
118 }

127 public static string? CreateAttrList(Style.Modifier? modifier)
128 {
129 return modifier?.DivideSingles()
130 .Select(x => x switch
131 {
132 Style.Modifier.Bold => "bold",
133 Style.Modifier.Italic => "italic",
134 Style.Modifier.Underline => "underline",
135 Style.Modifier.UnderlineWaved => "undercurl",
136 Style.Modifier.UnderlineDotted => "underdot",
137 Style.Modifier.UnderlineDashed => "underdash",
138 Style.Modifier.Strikethrough => "strikethrough",
139 Style.Modifier.None => "NONE",
140 _ => null
141 })
142 .WhereNotNull()
143 .DefaultIfEmpty(null)
144 .Aggregate((x, y) => $"{x},{y}");
145 }
