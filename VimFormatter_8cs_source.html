32 protected override SingleTextContent Format(IEnumerable<IVimSourceItemBase> items, BuilderQuery query)
33 {
34 var metadata = query.GetMetadata();
35 var header = StdFormatterImpl.CreateHeader(metadata, "\"");
36 var footer = StdFormatterImpl.CreateFooter(metadata, "\"");
37 var body = CreateBody(items.ToArray());
38
39 return new SingleTextContent($"colors/{metadata.ThemeName}.vim",
40 string.Join('\n', header),
41 $"""
42 set background={(metadata.ThemeType == ThemeType.Light ? "light" : "dark")}
43 highlight clear
44 if exists('syntax_on')
45 syntax reset
46 endif
47 set t_Co=256
48 let g:colors_name = '{metadata.ThemeName ?? "sccg_default"}'
49 """,
50 string.Join('\n', body),
51 string.Join('\n', footer)
52 );
53 }

55 private static IEnumerable<string> CreateBody(IVimSourceItemBase[] items)
56 {
57 var sb = new StringBuilder();
58
59 void Set<T>(string name, T? value)
60 {
61 if (value is null) return;
62 switch (value)
63 {
64 case Color c:
65 if (c.IsDefault) return;
66 var code = c.IsNone ? "NONE" : c.HexCode;
67 sb.Append($" {name}={code}");
68 break;
69 case string s:
70 sb.Append($" {name}={s}");
71 break;
72 default:
73 throw new NotSupportedException($"NeovimFormatter does not support type {typeof(T).Name}");
74 }
75 }
76
77 // :hi
78 foreach (var item in items.OfType<IVimSourceItem>())
79 {
80 var formattable = item.Extract();
81
82 if (formattable.Link is not null)
83 {
84 yield return $"hi link {formattable.Name} {formattable.Link}";
85 continue;
86 }
87
88 sb.Clear();
89 sb.Append($"hi {formattable.Name}{(formattable.Default ? " default" : "")}");
90 Set("ctermfg", formattable.Style?.Foreground.TerminalColorCode);
91 Set("ctermbg", formattable.Style?.Background.TerminalColorCode);
92 Set("ctermul", formattable.Style?.Special.TerminalColorCode);
93 Set("cterm", CreateAttrList(formattable.Style?.Modifiers));
94 Set("guifg", formattable.Style?.Foreground);
95 Set("guibg", formattable.Style?.Background);
96 Set("guisp", formattable.Style?.Special);
97 Set("gui", CreateAttrList(formattable.Style?.Modifiers));
98
99 yield return sb.ToString();
100 }
101
102 // let g:* = []
103 var arrayItem = items.OfType<IVimArrayVariableSourceItem>()
104 .Select(x => x.Extract())
105 .GroupBy(x => x.Name);
106 foreach (var one in arrayItem)
107 {
108 var name = one.Key;
109 var data = one.OrderBy(x => x.Index)
110 .Select(x => (x.Value, x.Index, x.Length))
111 .ToArray();
112
113 if (data.Select(x => x.Length).Distinct().Count() != 1)
114 {
115 throw new InvalidDataException($"Vim-array: {name} has different length.");
116 }
117
118 var idx = data.Select(x => x.Index).Distinct().ToArray();
119 if (idx.Length != data.Length || idx.First() != 0 || idx.Last() != data.Length - 1)
120 {
121 throw new InvalidDataException("Vim-array: Index is not unique.");
122 }
123
124 var formattedValues = string.Join(", ", data.Select(x => $"'{x.Value}'"));
125 yield return $"let g:{name} = [{formattedValues}]";
126 }
127 }

143 public static string? CreateAttrList(Style.Modifier? modifier)
144 {
145 return modifier?.DivideSingles()
146 .Select(x => x switch
147 {
148 Style.Modifier.Bold => "bold",
149 Style.Modifier.Italic => "italic",
150 Style.Modifier.Underline => "underline",
151 Style.Modifier.UnderlineWaved => "undercurl",
152 Style.Modifier.UnderlineDotted => "underdot",
153 Style.Modifier.UnderlineDashed => "underdash",
154 Style.Modifier.Strikethrough => "strikethrough",
155 Style.Modifier.None => "NONE",
156 _ => null
157 })
158 .WhereNotNull()
159 .DefaultIfEmpty(null)
160 .Aggregate((x, y) => $"{x},{y}");
161 }
