21 protected override IEnumerable<Item> CollectItems()
22 {
23 var ids = _impl.Graph.TopologicalOrderList();
24
25 foreach (var id in ids)
26 {
27 var data = _impl.Store.Load(id).data;
28 var next = _impl.Graph.GetLink(id);
29
30 if (data is Group fromGroup && next is not null)
31 {
32 var to = _impl.Store.Load(next.Value).data;
33 switch (to)
34 {
35 case Group toGroup:
36 yield return new Item(fromGroup, toGroup);
37 break;
38 case Style style:
39 yield return new Item(fromGroup, style);
40 break;
41 case VimSyntaxGroupSource.Group vimSyntaxGroup:
42 yield return new Item(fromGroup, vimSyntaxGroup);
43 break;
44 }
45 }
46 }
47 }

56 protected void Link(Group from, VimSyntaxGroupSource.Group to)
57 {
58 var fromId = _impl.Store.Save(from);
