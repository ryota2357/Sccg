22 protected override IEnumerable<Item> CollectItems()
23 {
24 var ids = _impl.Graph.TopologicalOrderList();
25 var save = new Dictionary<Group, Style>();
26
27 foreach (var id in ids)
28 {
29 var data = _impl.Store.Load(id).data;
30 var next = _impl.Graph.GetLink(id);
31
32 if (data is Group fromGroup && next is not null)
33 {
34 var to = _impl.Store.Load(next.Value).data;
35 switch (to)
36 {
37 case Group toGroup:
38 Style? sty = save.ContainsKey(toGroup) ? save[toGroup] : null;
39 if (sty is not null)
40 {
41 save[fromGroup] = sty.Value;
42 }
43 yield return new Item(fromGroup, toGroup, sty);
44 break;
45 case Style style:
46 save[fromGroup] = style;
47 yield return new Item(fromGroup, style);
48 break;
49 case VimSyntaxGroupSource.Group vimSyntaxGroup:
50 yield return new Item(fromGroup, vimSyntaxGroup);
51 break;
52 }
53 }
54 }
55 }

64 protected void Link(Group from, VimSyntaxGroupSource.Group to)
