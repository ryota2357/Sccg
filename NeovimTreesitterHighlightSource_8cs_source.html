17 protected override IEnumerable<Item> CollectItems()
18 {
19 var ids = _impl.Graph.TopologicalOrderList();
20
21 foreach (var id in ids)
22 {
23 var data = _impl.Store.Load(id).data;
24 var next = _impl.Graph.GetLink(id);
25
26 if (data is Group fromGroup && next is not null)
27 {
28 var to = _impl.Store.Load(next.Value).data;
29 switch (to)
30 {
31 case Group toGroup:
32 yield return new Item(fromGroup, toGroup);
33 break;
34 case Style style:
35 yield return new Item(fromGroup, style);
36 break;
37 case VimSyntaxGroupSource.Group vimSyntaxGroup:
38 yield return new Item(fromGroup, vimSyntaxGroup);
39 break;
40 }
41 }
42 }
43 }

46
47 protected override void Link(Group from, Group to) => _impl.Link(from, to);

50 protected void Link(Group from, VimSyntaxGroupSource.Group to)
51 {
52 var fromId = _impl.Store.Save(from);
53 var toId = _impl.Store.Save(to);
54 var status = _impl.Graph.CreateLink(fromId, toId);
55 if (status == false)

57 Log.Warn($"Ignored duplicate. Link({from}, {to})");
58 }
