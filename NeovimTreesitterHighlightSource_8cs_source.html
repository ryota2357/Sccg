31 protected override IEnumerable<Item> CollectItems()
32 {
33 var ids = _impl.Graph.TopologicalOrderList();
34 var save = new Dictionary<FiletypedGroup, Style>();
35
36 foreach (var id in ids)
37 {
38 var data = _impl.Store.Load(id).data;
39 var next = _impl.Graph.GetLink(id);
40
41 if (data is FiletypedGroup fromGroup && next is not null)
42 {
43 var to = _impl.Store.Load(next.Value).data;
44 switch (to)
45 {
46 case FiletypedGroup toGroup:
47 Style? sty = save.ContainsKey(toGroup) ? save[toGroup] : null;
48 if (sty is not null)
49 {
50 save[fromGroup] = sty.Value;
51 }
52 yield return new Item(fromGroup.Deconstruct(), toGroup.Deconstruct(), sty);
53 break;
54 case Style style:
55 save[fromGroup] = style;
56 yield return new Item(fromGroup.Deconstruct(), style);
57 break;
58 case VimSyntaxGroupSource.Group vimSyntaxGroup:
59 yield return new Item(fromGroup.Deconstruct(), vimSyntaxGroup);
60 break;
61 }
62 }
63 }
