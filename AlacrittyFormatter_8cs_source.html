41 private static string CreateBody(List<Formattable> items)
42 {
43 items.Sort((a, b) =>
44 {
45 var min = Math.Min(a.Keys.Count, b.Keys.Count);
46 for (var i = 0; i < min; i++)
47 {
48 var cmp = string.Compare(a.Keys[i], b.Keys[i], StringComparison.Ordinal);
49 if (cmp != 0)
50 {
51 return cmp;
52 }
53 }
54 return a.Keys.Count.CompareTo(b.Keys.Count);
55 });
56 var sb = new StringBuilder();
57 for (var i = 0; i < items.Count; i++)
58 {
59 for (var depth = 0; depth < items[i].Keys.Count; depth++)
60 {
61 if (i > 0 && depth < items[i - 1].Keys.Count && items[i - 1].Keys[depth] == items[i].Keys[depth])
62 {
63 continue;
64 }
65 sb.Append(' ', depth * 2);
66 sb.Append(items[i].Keys[depth]);
67 sb.Append(':');
68 if (depth == items[i].Keys.Count - 1)
69 {
70 sb.Append(' ');
71 sb.Append(items[i].Value);
72 }
73 sb.AppendLine();
74 }
75 }
76 return sb.ToString()[..^1]; // remove last newline
77 }
