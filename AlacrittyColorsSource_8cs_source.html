20 var ids = _impl.Graph.TopologicalOrderList();
21
22 Dictionary<int, Color> cache = new();
23 foreach (var id in ids)
24 {
25 if (!_impl.Store.TryLoad(id, out Group group)) continue;
26 if (!_impl.Graph.TryGetLink(id, out var next)) continue;
