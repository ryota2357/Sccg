21 protected override IEnumerable<Item> CollectItems()
22 {
23 var ids = _impl.Graph.TopologicalOrderList();
24 var save = new Dictionary<Group, Style>();
25
26 foreach (var id in ids)
27 {
28 var data = _impl.Store.Load(id).data;
29 var next = _impl.Graph.GetLink(id);
30
31 if (data is Group fromGroup && next is not null)
32 {
33 var to = _impl.Store.Load(next.Value).data;
34 switch (to)
35 {
36 case Group toGroup:
37 Style? sty = save.ContainsKey(toGroup) ? save[toGroup] : null;
38 if (sty is not null)
39 {
40 save[fromGroup] = sty.Value;
41 }
42 yield return new Item(fromGroup, toGroup, sty);
43 break;
44 case Style style:
45 save[fromGroup] = style;
46 yield return new Item(fromGroup, style);
47 break;
48 }
49 }
50 }
51 }
