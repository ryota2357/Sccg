16 protected override IEnumerable<Item> CollectItems()
17 {
18 var ids = _impl.Graph.TopologicalOrderList();
19
20 foreach (var id in ids)
21 {
22 var data = _impl.Store.Load(id).data;
23 var next = _impl.Graph.GetLink(id);
24
25 if (data is Group fromGroup && next is not null)
26 {
27 var to = _impl.Store.Load(next.Value).data;
28 switch (to)
29 {
30 case Group toGroup:
31 yield return new Item(fromGroup, toGroup);
32 break;
33 case Style style:
34 yield return new Item(fromGroup, style);
35 break;
36 }
37 }
38 }
39 }
40
41 protected override void Set(Group group, Style style) => _impl.Set(group, style);
42
43 protected override void Link(Group from, Group to) => _impl.Link(from, to);
