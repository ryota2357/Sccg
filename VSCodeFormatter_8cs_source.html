59 private static SingleTextContent CreateThemeJson(IVSCodeSourceItemBase[] items, Metadata metadata)
60 {
61 var sb = new StringBuilder();
62 sb.AppendLine("{");
63 sb.AppendLine($" \"name\": \"{metadata.Context.Get(_contextKey_vscode_displayName, metadata.ThemeName) ?? "unknown"}\",");
64 sb.AppendLine($" \"type\": \"{(metadata.ThemeType == ThemeType.Light ? "light" : "dark")}\",");
65
66 // colors
67 sb.AppendLine(" \"colors\": {");
68 var colors = items.OfType<IVSCodeColorSourceItem>()
69 .ToArray();
70 foreach (var item in colors)
71 {
72 var data = item.Extract();
73 sb.Append(" ");
74 sb.AppendLine($"\"{data.Name}\": \"{data.Color.HexCode}\",");
75 }
76 if (colors.Any())
77 {
78 sb.Remove(sb.Length - 2, 1); // remove last comma
79 }
80 sb.AppendLine(" },");
81
82 // tokenColors
83 sb.AppendLine(" \"tokenColors\": [");
84 var tokenColors = items.OfType<IVSCodeTokenColorSourceItem>()
85 .Select(i => i.Extract())
86 .GroupBy(i => i.Style)
87 .ToArray();
88 foreach (var item in tokenColors)
89 {
90 sb.AppendLine(" {");
91 if (item.Count() == 1)
92 {
93 var scope = item.First().Name;
94 sb.AppendLine($" \"scope\": \"{scope}\",");
95 }
96 else
97 {
98 sb.AppendLine(" \"scope\": [");
99 foreach (var scope in item)
100 {
101 sb.AppendLine($" \"{scope.Name}\",");
102 }
103 sb.Remove(sb.Length - 2, 1); // remove last comma
104 sb.AppendLine(" ],");
105 }
106 sb.AppendLine(" \"settings\": {");
107 var style = item.Key;
108 var hasSettings = false;
109 if (style.Foreground.HasHexCode())
110 {
111 sb.AppendLine($" \"foreground\": \"{style.Foreground.HexCode}\",");
112 hasSettings = true;
113 }
114 // NOTE: VSCode does not support `background` for `tokenColors`
115 if (!style.Modifiers.Contains(Style.Modifier.Default) && !style.Modifiers.Contains(Style.Modifier.None))
116 {
117 // NOTE: only support `bold`, `strikethrough`, `underline`, `italic`
118 var styles = new List<string>();
119 if (style.Modifiers.Contains(Style.Modifier.Bold)) styles.Add("bold");
120 if (style.Modifiers.Contains(Style.Modifier.Strikethrough)) styles.Add("strikethrough");
121 if (style.Modifiers.Contains(Style.Modifier.Underline)) styles.Add("underline");
122 if (style.Modifiers.Contains(Style.Modifier.Italic)) styles.Add("italic");
123 if (styles.Count != 0)
124 {
125 var fontStyle = string.Join(" ", styles);
126 sb.AppendLine($" \"fontStyle\": \"{fontStyle}\",");
127 hasSettings = true;
128 }
129 }
130 if (hasSettings)
131 {
132 sb.Remove(sb.Length - 2, 1); // remove last comma
133 }
134 sb.AppendLine(" }");
135 sb.AppendLine(" },");
136 }
137 if (tokenColors.Any())
138 {
139 sb.Remove(sb.Length - 2, 1); // remove last comma
140 }
141 sb.AppendLine(" ]");
142
143 sb.Append('}');
144 return new SingleTextContent($"themes/{metadata.ThemeName}-color-theme.json", sb.ToString());
145 }
