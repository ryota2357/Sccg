61  void Set<T>(string name, T? value)
62  {
63  if (value is null) return;
64  switch (value)
65  {
66  case bool b:
67  // NOTE: Because of using nvim_set_hl(), don't have to set false value. `false` is same as omitting.
68  if (!b) return;

76  case int i:

78  break;
79  case string s:
