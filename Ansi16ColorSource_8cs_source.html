53 private IEnumerable<Item> CollectItems()
54 {
55 var ids = _impl.Graph.TopologicalOrderList();
56 Dictionary<int, Color> cache = new();
57
58 foreach (var id in ids)
59 {
60 // skip until group
61 if (!_impl.Store.TryLoad(id, out Group group)) continue;
62
63 // group must have link, because it is set/linked by Set/Link method.
64 if (!_impl.Graph.TryGetLink(id, out var next))
65 {
66 throw new Exception($"Source item data is broken. Group {group} does not have link.");
67 }
68
69 if (_impl.Store.TryLoad(next.Value, out Color col) || cache.TryGetValue(next.Value, out col))
70 {
71 cache[id] = col;
72 yield return new Item(group, col);
73 }
74 else
75 {
76 // A links to B, but B is not Color or not a group that has Color.
77 throw new InvalidDataException($"Group {group} does not have specific color.");
78 }
79 }
