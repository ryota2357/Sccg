35 protected override IEnumerable<Item> CollectItems()
36 {
37 var ids = _impl.Graph.TopologicalOrderList();
38 Dictionary<int, Color> cache = new();
39
40 foreach (var id in ids)
41 {
42 // skip until group
43 if (!_impl.Store.TryLoad(id, out Group group)) continue;
44
45 // group must have link, because it is set/linked by Set/Link method.
46 if (!_impl.Graph.TryGetLink(id, out var next))
47 {
48 throw new Exception($"Source item data is broken. Group {group} does not have link.");
49 }
50
51 if (_impl.Store.TryLoad(next.Value, out Color col) || cache.TryGetValue(next.Value, out col))
52 {
53 cache[id] = col;
54 yield return new Item(group, col, ItemTarget);
55 }
56 else
57 {
58 // A links to B, but B is not Color or not a group that has Color.
59 throw new InvalidDataException($"Group {group} does not have specific color.");
60 }
61 }
