22 protected override IEnumerable<Item> CollectItems()
23 {
24 var ids = _impl.Graph.TopologicalOrderList();
25 Dictionary<int, Color> cache = new();
26
27 foreach (var id in ids)
28 {
29 // skip until group
30 if (!_impl.Store.TryLoad(id, out Group group)) continue;
31
32 // group must have link, because it is set/linked by Set/Link method.
33 if (!_impl.Graph.TryGetLink(id, out var next))
34 {
35 throw new Exception($"Source item data is broken. Group {group} does not have link.");
36 }
37
38 if (_impl.Store.TryLoad(next.Value, out Color col) || cache.TryGetValue(next.Value, out col))
39 {
40 cache[id] = col;
41 yield return new Item(group, col);
42 }
43 else
44 {
45 // A links to B, but B is not Color or not a group that has Color.
46 throw new InvalidDataException($"Group {group} does not have specific color.");
47 }
48 }
