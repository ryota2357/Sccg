20 protected override IEnumerable<Item> CollectItems()
21 {
22 var ids = _impl.Graph.TopologicalOrderList();
23 var save = new Dictionary<Group, Style>();
24
25 foreach (var id in ids)
26 {
27 var data = _impl.Store.Load(id).data;
28 var next = _impl.Graph.GetLink(id);
29
30 if (data is Group fromGroup && next is not null)
31 {
32 var to = _impl.Store.Load(next.Value).data;
33 switch (to)
34 {
35 case Group toGroup:
36 Style? sty = save.ContainsKey(toGroup) ? save[toGroup] : null;
37 if (sty is not null)
38 {
39 save[fromGroup] = sty.Value;
40 }
41 yield return new Item(fromGroup, toGroup, sty);
42 break;
43 case Style style:
44 save[fromGroup] = style;
45 yield return new Item(fromGroup, style);
46 break;
47 }
48 }
49 }
50 }
